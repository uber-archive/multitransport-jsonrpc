function formatMessage(obj, eventEmitter) {
    var str = JSON.stringify(obj);
    var strlen = Buffer.byteLength(str);
    if(eventEmitter) eventEmitter.emit('outMessage', obj, strlen);
    var buf = new Buffer(4 + strlen);
    buf.writeUInt32BE(strlen, 0);
    buf.write(str, 4, strlen, 'utf8');
    return buf;
}

Since all messages start with a length prefix and the "current" message is the first in the buffers array, we can determine the message length just by the first buffer in the array. This technically assumes that a buffer is at least 4 bytes large, but that should be a safe assumption.

function getMessageLen(buffers) {
    if(buffers[0] && buffers[0].length >= 4) {
        return buffers[0].readUInt32BE(0);
    } else {
        return 0;
    }
}

function min() {

        return (val < curr) ? val : curr;
